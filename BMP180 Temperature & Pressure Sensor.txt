/*
  Project: BMP180 Temperature & Pressure Monitor
  Board  : Arduino / ESP32 / ESP8266
  Sensor : BMP180 (I2C)

  This program reads temperature and pressure
  from the BMP180 sensor and displays the values
  on the Serial Monitor.
*/

#include <Wire.h>
#include <Adafruit_BMP085.h>

// Create sensor object
Adafruit_BMP085 bmp;

void setup() {
  Serial.begin(9600);
  Serial.println("Initializing BMP180 Sensor...");

  // Initialize BMP180
  if (!bmp.begin()) {
    Serial.println("BMP180 sensor not detected!");
    Serial.println("Check wiring and I2C connection.");
    while (1); // Stop program
  }

  Serial.println("BMP180 sensor detected successfully!");
}

void loop() {
  float temperature = bmp.readTemperature();
  long pressure = bmp.readPressure();

  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.println(" Â°C");

  Serial.print("Pressure   : ");
  Serial.print(pressure);
  Serial.println(" Pa");

  Serial.println("-------------------------");
  delay(1000); // Read data every 1 second
}
